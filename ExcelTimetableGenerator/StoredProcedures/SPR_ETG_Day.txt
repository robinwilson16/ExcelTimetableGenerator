CREATE PROCEDURE [dbo].[SPR_ETG_Day]
AS
BEGIN
	SET NOCOUNT ON;

	SELECT
		DayID = CAST ( CNT.RowNum AS INT ),
		DayName = DATENAME ( WEEKDAY, CNT.RowNum - 1 ),
		Slots = SLT.NumSlots
	FROM (
		SELECT
			RowNum = 
				ROW_NUMBER () OVER ( 
					ORDER BY
						OBJ.object_id
				)
		FROM sys.all_objects OBJ
	) CNT
	LEFT JOIN ETG_DaySlot SLT
		ON SLT.DaySlotID = CNT.RowNum
	WHERE
		CNT.RowNum <= 5
	ORDER BY
		CNT.RowNum
END
